    Container(vault, "HashiCorp Vault / Key Vault", "Secrets Manager & KMS", "Guarda chaves privadas JWT, connection strings criptografadas do PostgreSQL e credenciais do RabbitMQ/Redis.")

    Container(api_lancamentos, "API de Lançamentos", ".NET 8 / mTLS", "Injetada com Managed Identity / Vault Agent. Lê credenciais do Vault no startup; sem segredos estáticos no appsettings.")
    ContainerDb(db_lancamentos, "Banco Lançamentos", "PostgreSQL 17 (mTLS)", "Exige autenticação forte e conexão cifrada SSL/TLS com credenciais dinâmicas.")

    ContainerQueue(rabbitmq, "Broker RabbitMQ", "RabbitMQ 3.13 (AMQPS)", "Conexões internas criptografadas com mTLS e rotação de usuários do Vault.")

    Container(api_consolidado, "API de Consolidado", ".NET 8 / mTLS", "Valida JWT assinado via JWKS do IdP e obtém credenciais do banco/Redis via Vault em tempo de execução.")
    ContainerDb(db_consolidado, "Banco Consolidado", "PostgreSQL 17 (mTLS)", "Acesso restrito à API de Consolidado via conexões cifradas.")
    ContainerDb(redis, "Cache Redis", "Redis 7 (TLS + AUTH)", "Protegido por senha forte mantida no Vault e conexão cifrada TLS.")
}
